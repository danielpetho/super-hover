  ts={`
import { createSuperHover } from "super-hover";

const list = document.querySelector("#list");
const superHover = createSuperHover({ root: list ?? undefined });

for (const row of document.querySelectorAll("#list li")) {
  row.setAttribute("data-super-hover", "");
  row.className =
    "rounded px-3 py-2 transition-colors data-[super-hover-active]:bg-neutral-100";
}

// later: superHover();
`}

  svelteLanguage="svelte"
  react={`
import { useSuperHoverRef } from "super-hover/react";

export function Example() {
  const rootRef = useSuperHoverRef({
    onEnter: (e) => {
      console.log("enter:", e.target);
    },
    onLeave: (e) => {
      console.log("leave:", e.target);
    },
  });

  const items = ["Alpha", "Beta", "Gamma"];

  return (
    <ul ref={rootRef} className="space-y-1">
      {items.map((item) => (
        <li
          key={item}
          data-super-hover
          className="rounded px-3 py-2 transition-colors data-[super-hover-active]:bg-neutral-100"
        >
          {item}
        </li>
      ))}
    </ul>
  );
}
`}
  ts={`
import { createSuperHover } from "super-hover";

const root = document.querySelector("#list")!;
const onEnter = (e: Event) => {
  console.log("enter:", e.target);
};
const onLeave = (e: Event) => {
  console.log("leave:", e.target);
};

root.addEventListener("superhoverenter", onEnter);
root.addEventListener("superhoverleave", onLeave);

const stop = createSuperHover({ root });

/* cleanup: remove listeners and call stop() */
`}
  vue={`
<script setup lang="ts">
  import { useSuperHover } from "super-hover/vue";

  const items = ["Alpha", "Beta", "Gamma"];

  const rootRef = useSuperHover({
    onEnter: (e: Event) => {
      console.log("enter:", e.target);
    },
    onLeave: (e: Event) => {
      console.log("leave:", e.target);
    },
  });
</script>

<template>
  <ul ref="rootRef" class="space-y-1">
    <li
      v-for="item in items"
      :key="item"
      data-super-hover
      class="rounded px-3 py-2 transition-colors data-[super-hover-active]:bg-neutral-100"
    >
      {{ item }}
    </li>
  </ul>
</template>
`}
  svelte={`
<script lang="ts">
  import { superHover } from "super-hover/svelte";

  const items = ["Alpha", "Beta", "Gamma"];
</script>

<ul
  class="space-y-1"
  use:superHover={{
    onEnter: (e: Event) => {
      console.log("enter:", e.target);
    },
    onLeave: (e: Event) => {
      console.log("leave:", e.target);

>
